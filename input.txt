from pundit.base import PunditBase
from pundit.condition import ConditionGroup
from pundit import Pundit

base = PunditBase()
base.add_structure("designation", "rank")
base.structure -> {"designation":{}, "rank"{}}


conditions = ConditionGroup()
conditions.add_condition("chief","IN", "designation")
conditions.add_condition("1","IN","rank", "Success")
conditions.response("success", "failed") -> response if satisfied


pundit = Pundit(base, conditions)

pundit.evaluate('chief technology officer', '2')

>>> Success

pundit.evaluate('sofware engineer')

>>> Success

pundit.evaluate('sofware')

>>> Failed
