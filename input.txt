pundit = Pundit('desc')
pundit.add_input(1, "Chief Technology Officer", 'str')
pundit.add_input(1, "VP of Technology", 'str')
pundit.add_input(1, "Chief Operations Officer", 'str')
pundit.add_input(2, "Software Engineer", 'str')

pundit.input_preprocess(1, 'lower')

condition.add_condition('chief', 'IN', 1)
condition.add_condition('software', 'IN', 2)


pundit.evaluate(1, condition)
